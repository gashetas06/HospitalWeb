--  CITAS

CREATE TABLE Cita (
    idCita           INT           PRIMARY KEY IDENTITY(1,1),
    idPaciente       INT           NOT NULL,
    idMedico         INT           NOT NULL,
    idEmpleadoRegistra INT         NULL,   -- quién agendó la cita
    fecha            DATE          NOT NULL,
    hora             TIME          NOT NULL,
    duracionMinutos  TINYINT       NOT NULL DEFAULT 30,
    tipoCita         NVARCHAR(50)  NOT NULL DEFAULT 'Consulta'
                         CHECK (tipoCita IN ('Consulta','Seguimiento','Urgencia','Procedimiento','Teleconsulta')),
    estado           NVARCHAR(30)  NOT NULL DEFAULT 'Programada'
                         CHECK (estado IN ('Programada','Confirmada','En curso','Completada','Cancelada','No asistió')),
    motivoCancelacion NVARCHAR(255) NULL,
    notas            NVARCHAR(500) NULL,
    fechaCreacion    DATETIME      NOT NULL DEFAULT GETDATE(),

    CONSTRAINT FK_Cita_Medico     FOREIGN KEY (idMedico)           REFERENCES Medico(idMedico),
    CONSTRAINT FK_Cita_Empleado   FOREIGN KEY (idEmpleadoRegistra) REFERENCES Empleado(idEmpleado)
);
GO

-- SP: Agendar una cita validando disponibilidad
CREATE OR ALTER PROCEDURE sp_AgendarCita
    @idPaciente       INT,
    @idMedico         INT,
    @fecha            DATE,
    @hora             TIME,
    @tipoCita         NVARCHAR(50) = 'Consulta',
    @notas            NVARCHAR(500) = NULL,
    @idEmpleado       INT = NULL
AS
BEGIN
    SET NOCOUNT ON;

    -- 1. Verificar que el médico esté activo
    IF NOT EXISTS (SELECT 1 FROM Medico WHERE idMedico = @idMedico AND activo = 1)
    BEGIN
        RAISERROR('El médico no existe o no está activo.', 16, 1);
        RETURN;
    END

    -- 2. Verificar que el paciente esté activo
    IF NOT EXISTS (SELECT 1 FROM Paciente WHERE idPaciente = @idPaciente AND activo = 1)
    BEGIN
        RAISERROR('El paciente no existe o no está activo.', 16, 1);
        RETURN;
    END

    -- 3. Verificar que no exista otra cita activa para ese médico en esa fecha/hora
    IF EXISTS (
        SELECT 1 FROM Cita
        WHERE idMedico = @idMedico
          AND fecha = @fecha
          AND hora  = @hora
          AND estado NOT IN ('Cancelada', 'No asistió')
    )
    BEGIN
        RAISERROR('El médico ya tiene una cita programada en esa fecha y hora.', 16, 1);
        RETURN;
    END

    -- 4. Verificar que el médico atiende ese día de la semana
    DECLARE @diaSemana TINYINT = DATEPART(WEEKDAY, @fecha);
    -- Ajuste: DATEPART devuelve 1=Dom en SQL Server; convertimos a 1=Lun
    SET @diaSemana = CASE @diaSemana WHEN 1 THEN 7 ELSE @diaSemana - 1 END;

    IF NOT EXISTS (
        SELECT 1 FROM HorarioMedico
        WHERE idMedico   = @idMedico
          AND diaSemana  = @diaSemana
          AND @hora BETWEEN horaInicio AND horaFin
    )
    BEGIN
        RAISERROR('El médico no tiene horario disponible en ese día u hora.', 16, 1);
        RETURN;
    END

    -- 5. Insertar la cita
    INSERT INTO Cita (idPaciente, idMedico, idEmpleadoRegistra, fecha, hora, tipoCita, notas)
    VALUES (@idPaciente, @idMedico, @idEmpleado, @fecha, @hora, @tipoCita, @notas);

    SELECT SCOPE_IDENTITY() AS idCitaGenerada;
END;
GO

-- SP: Cancelar una cita
CREATE OR ALTER PROCEDURE sp_CancelarCita
    @idCita           INT,
    @motivoCancelacion NVARCHAR(255) = NULL
AS
BEGIN
    SET NOCOUNT ON;

    IF NOT EXISTS (SELECT 1 FROM Cita WHERE idCita = @idCita)
    BEGIN
        RAISERROR('La cita no existe.', 16, 1);
        RETURN;
    END

    IF EXISTS (SELECT 1 FROM Cita WHERE idCita = @idCita AND estado IN ('Completada','Cancelada'))
    BEGIN
        RAISERROR('La cita ya fue completada o cancelada.', 16, 1);
        RETURN;
    END

    UPDATE Cita
    SET estado = 'Cancelada', motivoCancelacion = @motivoCancelacion
    WHERE idCita = @idCita;

    PRINT 'Cita ' + CAST(@idCita AS NVARCHAR) + ' cancelada correctamente.';
END;
GO

-- SP: Dar de alta a un paciente internado y liberar su cama
CREATE OR ALTER PROCEDURE sp_AltaPaciente
    @idInternacion    INT,
    @diagnosticoEgreso NVARCHAR(500) = NULL,
    @condicionEgreso  NVARCHAR(50)   = 'Alta médica'
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;

    DECLARE @idCama INT;
    SELECT @idCama = idCama FROM Internacion WHERE idInternacion = @idInternacion;

    IF @idCama IS NULL
    BEGIN
        ROLLBACK;
        RAISERROR('Internación no encontrada.', 16, 1);
        RETURN;
    END

    -- Registrar egreso
    UPDATE Internacion
    SET fechaEgreso      = GETDATE(),
        diagnosticoEgreso = @diagnosticoEgreso,
        condicionEgreso   = @condicionEgreso
    WHERE idInternacion = @idInternacion;

    -- Liberar cama
    UPDATE Cama SET estado = 'Libre' WHERE idCama = @idCama;

    COMMIT;
    PRINT 'Alta registrada y cama liberada.';
END;
GO

-- SP: Registrar movimiento de inventario y actualizar stock
CREATE OR ALTER PROCEDURE sp_RegistrarMovimientoInventario
    @idInventario    INT,
    @tipoMovimiento  NVARCHAR(20),
    @cantidad        INT,
    @motivo          NVARCHAR(255) = NULL,
    @idEmpleado      INT           = NULL
AS
BEGIN
    SET NOCOUNT ON;
    BEGIN TRANSACTION;

    IF @tipoMovimiento = 'Salida'
    BEGIN
        -- Verificar stock suficiente
        IF (SELECT cantidadDisponible FROM Inventario WHERE idInventario = @idInventario) < @cantidad
        BEGIN
            ROLLBACK;
            RAISERROR('Stock insuficiente para registrar la salida.', 16, 1);
            RETURN;
        END
        UPDATE Inventario
        SET cantidadDisponible    = cantidadDisponible - @cantidad,
            fechaUltimaEntrada    = GETDATE()
        WHERE idInventario = @idInventario;
    END
    ELSE IF @tipoMovimiento = 'Entrada'
    BEGIN
        UPDATE Inventario
        SET cantidadDisponible    = cantidadDisponible + @cantidad,
            fechaUltimaEntrada    = GETDATE()
        WHERE idInventario = @idInventario;
    END
    ELSE -- Ajuste directo (sobrescribe)
    BEGIN
        UPDATE Inventario
        SET cantidadDisponible    = @cantidad,
            fechaUltimaEntrada    = GETDATE()
        WHERE idInventario = @idInventario;
    END

    INSERT INTO MovimientoInventario (idInventario, idEmpleado, tipoMovimiento, cantidad, motivo)
    VALUES (@idInventario, @idEmpleado, @tipoMovimiento, @cantidad, @motivo);

    COMMIT;
END;
GO

ON dbo.Inventario

    SET NOCOUNT ON;

        'ALERTA: Producto [' + i.nombreProducto + '] bajo nivel critico. Stock: '

--  TRIGGER 2: Validar y ocupar cama al registrar una internación
-- ============================================================
USE HospitalDB;
GO

IF OBJECT_ID('dbo.trg_AsignarCamaInternacion', 'TR') IS NOT NULL
    DROP TRIGGER dbo.trg_AsignarCamaInternacion;
GO

USE HospitalDB;
GO

CREATE TRIGGER dbo.trg_AsignarCamaInternacion
ON dbo.Internacion
AFTER INSERT
AS
BEGIN
    SET NOCOUNT ON;

    -- Manejar inserción de múltiples filas correctamente
    IF EXISTS (
        SELECT 1
        FROM inserted i
        JOIN dbo.Cama c ON i.idCama = c.idCama
        WHERE c.estado <> 'Libre'
    )
    BEGIN
        RAISERROR('Una o más camas seleccionadas no están disponibles.', 16, 1);
        ROLLBACK TRANSACTION;
        RETURN;
    END

    -- Ocupar todas las camas de las filas insertadas
    UPDATE dbo.Cama
    SET estado = 'Ocupada'
    WHERE idCama IN (SELECT idCama FROM inserted);
END;
GO

-- ============================================================
--  TRIGGER 3: Auditoría automática de cambios en Cita
-- ============================================================
USE HospitalDB;
GO

IF OBJECT_ID('dbo.trg_AuditoriaCita', 'TR') IS NOT NULL
    DROP TRIGGER dbo.trg_AuditoriaCita;
GO

USE HospitalDB;
GO

CREATE TRIGGER dbo.trg_AuditoriaCita
ON dbo.Cita
AFTER INSERT, UPDATE, DELETE
AS
BEGIN
    SET NOCOUNT ON;

    -- UPDATE: existen filas en inserted Y en deleted
    IF EXISTS (SELECT 1 FROM inserted) AND EXISTS (SELECT 1 FROM deleted)
    BEGIN
        INSERT INTO dbo.LogAuditoria (tabla, operacion, idRegistro, descripcion)
        SELECT
            'Cita',
            'UPDATE',
            i.idCita,
            'Estado anterior: ' + ISNULL(d.estado, 'N/A') + ' -> Nuevo: ' + ISNULL(i.estado, 'N/A')
        FROM inserted i
        JOIN deleted d ON i.idCita = d.idCita;
    END

    -- INSERT: solo existen filas en inserted
    ELSE IF EXISTS (SELECT 1 FROM inserted)
    BEGIN
        INSERT INTO dbo.LogAuditoria (tabla, operacion, idRegistro, descripcion)
        SELECT
            'Cita',
            'INSERT',
            idCita,
            'Cita creada para paciente ID: ' + CAST(idPaciente AS NVARCHAR(10))
        FROM inserted;
    END

    -- DELETE: solo existen filas en deleted
    ELSE
    BEGIN
        INSERT INTO dbo.LogAuditoria (tabla, operacion, idRegistro, descripcion)
        SELECT
            'Cita',
            'DELETE',
            idCita,
            'Cita eliminada. Paciente ID: ' + CAST(idPaciente AS NVARCHAR(10))
        FROM deleted;
    END
END;
GO

-- Vista: Agenda diaria completa
CREATE OR ALTER VIEW vw_AgendaDiaria AS
SELECT
    c.idCita,
    c.fecha,
    c.hora,
    c.duracionMinutos,
    c.tipoCita,
    c.estado,
    p.cedula            AS cedulaPaciente,
    p.nombre + ' ' + p.apellido AS nombrePaciente,
    p.telefono          AS telefonoPaciente,
    m.nombre + ' ' + m.apellido AS nombreMedico,
    e.nombre            AS especialidad,
    c.notas
FROM Cita c
JOIN Paciente   p  ON c.idPaciente  = p.idPaciente
JOIN Medico     m  ON c.idMedico    = m.idMedico
JOIN Especialidad e ON m.idEspecialidad = e.idEspecialidad;
GO

-- Vista: Camas con su estado actual e información de internación activa
CREATE OR ALTER VIEW vw_EstadoCamas AS
SELECT
    ca.idCama,
    ca.numero,
    ca.estado           AS estadoCama,
    d.nombre            AS departamento,
    th.descripcion      AS tipoHabitacion,
    th.costoDiario,
    p.nombre + ' ' + p.apellido AS pacienteActual,
    i.fechaIngreso,
    i.motivoIngreso
FROM Cama ca
JOIN Departamento   d  ON ca.idDepartamento = d.idDepartamento
JOIN TipoHabitacion th ON ca.idTipo         = th.idTipo
LEFT JOIN Internacion i  ON ca.idCama        = i.idCama AND i.fechaEgreso IS NULL
LEFT JOIN Paciente   p  ON i.idPaciente      = p.idPaciente;
GO

-- Vista: Inventario en estado crítico
CREATE OR ALTER VIEW vw_InventarioCritico AS
SELECT
    inv.idInventario,
    cat.nombre              AS categoria,
    inv.nombreProducto,
    inv.cantidadDisponible,
    inv.nivelCritico,
    inv.cantidadDisponible - inv.nivelCritico AS deficit,
    inv.proveedor
FROM Inventario inv
JOIN CategoriaInventario cat ON inv.idCategoria = cat.idCategoria
WHERE inv.cantidadDisponible <= inv.nivelCritico;
GO

-- Vista: Historial de citas por paciente
CREATE OR ALTER VIEW vw_HistorialCitasPaciente AS
SELECT
    p.idPaciente,
    p.nombre + ' ' + p.apellido AS paciente,
    COUNT(c.idCita)             AS totalCitas,
    SUM(CASE WHEN c.estado = 'Completada'  THEN 1 ELSE 0 END) AS completadas,
    SUM(CASE WHEN c.estado = 'Cancelada'   THEN 1 ELSE 0 END) AS canceladas,
    SUM(CASE WHEN c.estado = 'No asistió'  THEN 1 ELSE 0 END) AS noAsistio,
    MAX(c.fecha)                AS ultimaCita
FROM Paciente p
LEFT JOIN Cita c ON p.idPaciente = c.idPaciente
GROUP BY p.idPaciente, p.nombre, p.apellido;
GO
